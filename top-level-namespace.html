def slack(text)
	return STDERR.print("No notify text") unless text

	url = ENV['WEBHOOK_URL']
	if url
		Slack::Notifier.new(url).ping(text)
	else
		STDERR.print("'WEBHOOK_URL' is undefined.")
	end

end
