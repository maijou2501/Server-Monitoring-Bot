def get_status_code(uri)
	time = ENV["CHECK_TIMEOUT"] || 5
	timeout(time){ Net::HTTP.get_response(URI.parse(uri)).code }
end

def notify(text)
	url = ENV['WEBHOOK_URL']
	if url
		Slack::Notifier.new(url).ping(text)
	else
		STDERR.print("'WEBHOOK_URL' is undefined.")
		return 1
	end

end

def ping(host)
	pingmon = Net::Ping::External.new(host)
	pingmon.timeout = ENV["CHECK_TIMEOUT"] || 5
	if pingmon.ping?
		TRUE
	else
		FALSE
	end
end
